import java.util.List;
import java.util.Objects;

public class PolicyDataResponse {
    private Short applicationYr;
    private String policyNum;
    private List<Group> list;

    // getters and setters (if needed)
    public Short getApplicationYr() {
        return applicationYr;
    }

    public void setApplicationYr(Short applicationYr) {
        this.applicationYr = applicationYr;
    }

    public String getPolicyNum() {
        return policyNum;
    }

    public void setPolicyNum(String policyNum) {
        this.policyNum = policyNum;
    }

    public List<Group> getList() {
        return list;
    }

    public void setList(List<Group> list) {
        this.list = list;
    }

    // equals
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof PolicyDataResponse)) return false;
        PolicyDataResponse that = (PolicyDataResponse) o;
        return Objects.equals(applicationYr, that.applicationYr) &&
               Objects.equals(policyNum, that.policyNum) &&
               Objects.equals(list, that.list);
    }

    // hashCode
    @Override
    public int hashCode() {
        return Objects.hash(applicationYr, policyNum, list);
    }

    // optional: toString
    @Override
    public String toString() {
        return "PolicyDataResponse[" +
               "applicationYr=" + applicationYr +
               ", policyNum=" + policyNum +
               ", list=" + list +
               ']';
    }
}
